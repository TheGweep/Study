Сортировка обменом по возрастанию (методом «пузырька») 

	mov c,5  	; c = 5
next:	lxi h,arr	; помещаем адрес на массив в пару HL (Запись в регистры H, L)
	mov d,c		; d = c
m1:	mov a,m 	; помещаем первый элемент массива в аккумулятор a = arr[d]
 
	inx h   	; переходим к следующему элементу массива,	arr[d+1] (inx - добавление единицы)
 
	sub a,m		; a = a - m, arr[d] - arr[d+1]
	jc m2		; если получилось положительное число значит a < m , arr[d] < arr[d+1], заканчиваем итерацию (jc - если происходит перенос) (JC	CF = 1	если перенос)
			
			; иначе меняем местами

	add m		; a = a + m
	mov b,m		; b = m , b = arr[d+1]
	mov m,a		; m = a , arr[d+1] = arr[d]
	dcx h		; переходим к предыдущему элементу, arr[d] (dcx - возврат к предыдущему элементу)
	mov m,b		; m = b, arr[d] = b
	inx h		; возвращаемся на текущий элемент, arr[d+1] на следующей итерации просто arr[d]
m2:dcr d		; d++
	jnz m1		; если d != 0 , то продолжаем цикл перемещения  move (jnz != 0 ) (JNZ	ZF = 0	если нет нуля)
	dcr c		; c++
	jnz next 	; если c != 0 то продолжаем цикл next
 
 
	end		; выходим из программы
 
	arr db 2, 1, 3, 4, 5 ;заданный массив (db - 1 байт)

; c = cl, h = hl, d = dl, a = al,
